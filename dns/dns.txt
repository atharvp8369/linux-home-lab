# This setup was tested in a client-server environment using RHEL.
# DNS Server Configuration (BIND)

# Install package
dnf install bind -y

# Edit main config
vim /etc/named.conf

listen-on port 53 { 192.168.0.254; };
allow-query { 192.168.0.0/24; };

# Go to /var/named
cd /var/named

# Create zone files
cp -vp named.localhost forward.zone
cp -vp named.loopback reverse.zone

# Edit forward zone
vim forward.zone

$TTL 1D
@   IN  SOA skynet.com. root.server.skynet.com. (
        0       ; serial
        1D      ; refresh
        1H      ; retry
        1W      ; expire
        3H )    ; minimum

@       IN  NS      server.skynet.com.
server  IN  A       192.168.0.254
client1 IN  A       192.168.0.1
client2 IN  A       192.168.0.2
www     IN  CNAME   server.skynet.com.
@       IN  MX 10   server.skynet.com.

# Edit reverse zone
vim reverse.zone

$TTL 1D
@   IN  SOA skynet.com. root.server.skynet.com. (
        0
        1D
        1H
        1W
        3H )

@       IN  NS      server.skynet.com.
254     IN  PTR     server.skynet.com.
1       IN  PTR     client1.skynet.com.
2       IN  PTR     client2.skynet.com.

# Restart service
systemctl restart named
systemctl enable named

# Allow firewall
firewall-cmd --permanent --add-service=dns
firewall-cmd --reload


# -------- CLIENT SIDE --------

# Configure DNS on client
vim /etc/resolv.conf

search skynet.com
nameserver 192.168.0.254

# Test DNS
nslookup server.skynet.com
